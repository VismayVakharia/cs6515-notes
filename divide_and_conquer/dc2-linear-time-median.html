FastSelect(A, k):
	input: unsorted A and integer k where 1<=k<=n
	output: k-th smallest of A
	
	Break A into n/5 groups G1...Gn/5
	for i=1...n/5:
		sort Gi
		mi = median(Gi)
	Let S={m1...mn/5}
	p = FastSelect(S, n/10)
	
	Partition A into A(<p), A(=p), A(>p)
	if k<=|A(<p)|:
		then return(FastSelect(A(<p), k))
	if k>|A(<=p)|+|A(=p)|:
		then return(FastSelect(A(>p), k-|A(<p)|-|A(=p)|))
	else:
		return(p)
